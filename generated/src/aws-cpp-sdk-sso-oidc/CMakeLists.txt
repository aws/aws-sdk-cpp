add_project(aws-cpp-sdk-sso-oidc "C++ SDK for the AWS sso-oidc service" aws-cpp-sdk-core)

file(GLOB AWS_SSO-OIDC_HEADERS
    "include/aws/sso-oidc/*.h"
)

file(GLOB AWS_SSO-OIDC_MODEL_HEADERS
    "include/aws/sso-oidc/model/*.h"
)

file(GLOB AWS_SSO-OIDC_SOURCE
    "source/*.cpp"
)

file(GLOB AWS_SSO-OIDC_MODEL_SOURCE
    "source/model/*.cpp"
)

file(GLOB SSO-OIDC_UNIFIED_HEADERS
    ${AWS_SSO-OIDC_HEADERS}
    ${AWS_SSO-OIDC_MODEL_HEADERS}
)

file(GLOB SSO-OIDC_UNITY_SRC
    ${AWS_SSO-OIDC_SOURCE}
    ${AWS_SSO-OIDC_MODEL_SOURCE}
)

if(ENABLE_UNITY_BUILD)
    enable_unity_build("SSO-OIDC" SSO-OIDC_UNITY_SRC)
endif()

file(GLOB SSO-OIDC_SRC
    ${SSO-OIDC_UNIFIED_HEADERS}
    ${SSO-OIDC_UNITY_SRC}
)

if(WIN32)
    #if we are compiling for visual studio, create a sane directory tree.
    if(MSVC)
        source_group("Header Files\\aws\\sso-oidc" FILES ${AWS_SSO-OIDC_HEADERS})
        source_group("Header Files\\aws\\sso-oidc\\model" FILES ${AWS_SSO-OIDC_MODEL_HEADERS})
        source_group("Source Files" FILES ${AWS_SSO-OIDC_SOURCE})
        source_group("Source Files\\model" FILES ${AWS_SSO-OIDC_MODEL_SOURCE})
    endif(MSVC)
endif()

set(SSO-OIDC_INCLUDES
    "${CMAKE_CURRENT_SOURCE_DIR}/include/"
)

add_library(${PROJECT_NAME} ${SSO-OIDC_SRC})
add_library(AWS::${PROJECT_NAME} ALIAS ${PROJECT_NAME})

set_compiler_flags(${PROJECT_NAME})
set_compiler_warnings(${PROJECT_NAME})

if(USE_WINDOWS_DLL_SEMANTICS AND BUILD_SHARED_LIBS)
    target_compile_definitions(${PROJECT_NAME} PRIVATE "AWS_SSOOIDC_EXPORTS")
endif()

target_include_directories(${PROJECT_NAME} PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
    $<INSTALL_INTERFACE:include>)

target_link_libraries(${PROJECT_NAME} PRIVATE ${PLATFORM_DEP_LIBS} ${PROJECT_LIBS})


setup_install()

install (FILES ${AWS_SSO-OIDC_HEADERS} DESTINATION ${INCLUDE_DIRECTORY}/aws/sso-oidc)
install (FILES ${AWS_SSO-OIDC_MODEL_HEADERS} DESTINATION ${INCLUDE_DIRECTORY}/aws/sso-oidc/model)

do_packaging()
