add_project(aws-cpp-sdk-textract "C++ SDK for the AWS textract service" aws-cpp-sdk-core)

file(GLOB AWS_TEXTRACT_HEADERS
    "include/aws/textract/*.h"
)

file(GLOB AWS_TEXTRACT_MODEL_HEADERS
    "include/aws/textract/model/*.h"
)

file(GLOB AWS_TEXTRACT_SOURCE
    "source/*.cpp"
)

file(GLOB AWS_TEXTRACT_MODEL_SOURCE
    "source/model/*.cpp"
)

file(GLOB TEXTRACT_UNIFIED_HEADERS
    ${AWS_TEXTRACT_HEADERS}
    ${AWS_TEXTRACT_MODEL_HEADERS}
)

file(GLOB TEXTRACT_UNITY_SRC
    ${AWS_TEXTRACT_SOURCE}
    ${AWS_TEXTRACT_MODEL_SOURCE}
)

if(ENABLE_UNITY_BUILD)
    enable_unity_build("TEXTRACT" TEXTRACT_UNITY_SRC)
endif()

file(GLOB TEXTRACT_SRC
    ${TEXTRACT_UNIFIED_HEADERS}
    ${TEXTRACT_UNITY_SRC}
)

if(WIN32)
    #if we are compiling for visual studio, create a sane directory tree.
    if(MSVC)
        source_group("Header Files\\aws\\textract" FILES ${AWS_TEXTRACT_HEADERS})
        source_group("Header Files\\aws\\textract\\model" FILES ${AWS_TEXTRACT_MODEL_HEADERS})
        source_group("Source Files" FILES ${AWS_TEXTRACT_SOURCE})
        source_group("Source Files\\model" FILES ${AWS_TEXTRACT_MODEL_SOURCE})
    endif(MSVC)
endif()

set(TEXTRACT_INCLUDES
    "${CMAKE_CURRENT_SOURCE_DIR}/include/"
)

add_library(${PROJECT_NAME} ${TEXTRACT_SRC})
add_library(AWS::${PROJECT_NAME} ALIAS ${PROJECT_NAME})

set_compiler_flags(${PROJECT_NAME})
set_compiler_warnings(${PROJECT_NAME})

if(USE_WINDOWS_DLL_SEMANTICS AND BUILD_SHARED_LIBS)
    target_compile_definitions(${PROJECT_NAME} PRIVATE "AWS_TEXTRACT_EXPORTS")
endif()

target_include_directories(${PROJECT_NAME} PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
    $<INSTALL_INTERFACE:include>)

target_link_libraries(${PROJECT_NAME} PRIVATE ${PLATFORM_DEP_LIBS} ${PROJECT_LIBS})


setup_install()

install (FILES ${AWS_TEXTRACT_HEADERS} DESTINATION ${INCLUDE_DIRECTORY}/aws/textract)
install (FILES ${AWS_TEXTRACT_MODEL_HEADERS} DESTINATION ${INCLUDE_DIRECTORY}/aws/textract/model)

do_packaging()
