add_project(aws-cpp-sdk-eks "C++ SDK for the AWS eks service" aws-cpp-sdk-core)

file(GLOB AWS_EKS_HEADERS
    "include/aws/eks/*.h"
)

file(GLOB AWS_EKS_MODEL_HEADERS
    "include/aws/eks/model/*.h"
)

file(GLOB AWS_EKS_SOURCE
    "source/*.cpp"
)

file(GLOB AWS_EKS_MODEL_SOURCE
    "source/model/*.cpp"
)

file(GLOB EKS_UNIFIED_HEADERS
    ${AWS_EKS_HEADERS}
    ${AWS_EKS_MODEL_HEADERS}
)

file(GLOB EKS_UNITY_SRC
    ${AWS_EKS_SOURCE}
    ${AWS_EKS_MODEL_SOURCE}
)

if(ENABLE_UNITY_BUILD)
    enable_unity_build("EKS" EKS_UNITY_SRC)
endif()

file(GLOB EKS_SRC
    ${EKS_UNIFIED_HEADERS}
    ${EKS_UNITY_SRC}
)

if(WIN32)
    #if we are compiling for visual studio, create a sane directory tree.
    if(MSVC)
        source_group("Header Files\\aws\\eks" FILES ${AWS_EKS_HEADERS})
        source_group("Header Files\\aws\\eks\\model" FILES ${AWS_EKS_MODEL_HEADERS})
        source_group("Source Files" FILES ${AWS_EKS_SOURCE})
        source_group("Source Files\\model" FILES ${AWS_EKS_MODEL_SOURCE})
    endif(MSVC)
endif()

set(EKS_INCLUDES
    "${CMAKE_CURRENT_SOURCE_DIR}/include/"
)

add_library(${PROJECT_NAME} ${LIBTYPE} ${EKS_SRC})
add_library(AWS::${PROJECT_NAME} ALIAS ${PROJECT_NAME})

set_compiler_flags(${PROJECT_NAME})
set_compiler_warnings(${PROJECT_NAME})

if(USE_WINDOWS_DLL_SEMANTICS AND BUILD_SHARED_LIBS)
    target_compile_definitions(${PROJECT_NAME} PRIVATE "AWS_EKS_EXPORTS")
endif()

target_include_directories(${PROJECT_NAME} PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
    $<INSTALL_INTERFACE:include>)

target_link_libraries(${PROJECT_NAME} PRIVATE ${PLATFORM_DEP_LIBS} ${PROJECT_LIBS})


setup_install()

install (FILES ${AWS_EKS_HEADERS} DESTINATION ${INCLUDE_DIRECTORY}/aws/eks)
install (FILES ${AWS_EKS_MODEL_HEADERS} DESTINATION ${INCLUDE_DIRECTORY}/aws/eks/model)

if(PLATFORM_WINDOWS AND MSVC)
    file(GLOB NUGET_FILES "nuget/${PROJECT_NAME}.*.autopkg")
    install (FILES ${NUGET_FILES} DESTINATION nuget)
endif()

do_packaging()
