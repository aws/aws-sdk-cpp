add_project(aws-cpp-sdk-pinpoint "C++ SDK for the AWS pinpoint service" aws-cpp-sdk-core)

file(GLOB AWS_PINPOINT_HEADERS
    "include/aws/pinpoint/*.h"
)

file(GLOB AWS_PINPOINT_MODEL_HEADERS
    "include/aws/pinpoint/model/*.h"
)

file(GLOB AWS_PINPOINT_SOURCE
    "source/*.cpp"
)

file(GLOB AWS_PINPOINT_MODEL_SOURCE
    "source/model/*.cpp"
)

file(GLOB PINPOINT_UNIFIED_HEADERS
    ${AWS_PINPOINT_HEADERS}
    ${AWS_PINPOINT_MODEL_HEADERS}
)

file(GLOB PINPOINT_UNITY_SRC
    ${AWS_PINPOINT_SOURCE}
    ${AWS_PINPOINT_MODEL_SOURCE}
)

if(ENABLE_UNITY_BUILD)
    enable_unity_build("PINPOINT" PINPOINT_UNITY_SRC)
endif()

file(GLOB PINPOINT_SRC
    ${PINPOINT_UNIFIED_HEADERS}
    ${PINPOINT_UNITY_SRC}
)

if(WIN32)
    #if we are compiling for visual studio, create a sane directory tree.
    if(MSVC)
        source_group("Header Files\\aws\\pinpoint" FILES ${AWS_PINPOINT_HEADERS})
        source_group("Header Files\\aws\\pinpoint\\model" FILES ${AWS_PINPOINT_MODEL_HEADERS})
        source_group("Source Files" FILES ${AWS_PINPOINT_SOURCE})
        source_group("Source Files\\model" FILES ${AWS_PINPOINT_MODEL_SOURCE})
    endif(MSVC)
endif()

set(PINPOINT_INCLUDES
    "${CMAKE_CURRENT_SOURCE_DIR}/include/"
)

add_library(${PROJECT_NAME} ${LIBTYPE} ${PINPOINT_SRC})
add_library(AWS::${PROJECT_NAME} ALIAS ${PROJECT_NAME})

set_compiler_flags(${PROJECT_NAME})
set_compiler_warnings(${PROJECT_NAME})

if(USE_WINDOWS_DLL_SEMANTICS AND BUILD_SHARED_LIBS)
    target_compile_definitions(${PROJECT_NAME} PRIVATE "AWS_PINPOINT_EXPORTS")
endif()

target_include_directories(${PROJECT_NAME} PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
    $<INSTALL_INTERFACE:include>)

target_link_libraries(${PROJECT_NAME} PRIVATE ${PLATFORM_DEP_LIBS} ${PROJECT_LIBS})


setup_install()

install (FILES ${AWS_PINPOINT_HEADERS} DESTINATION ${INCLUDE_DIRECTORY}/aws/pinpoint)
install (FILES ${AWS_PINPOINT_MODEL_HEADERS} DESTINATION ${INCLUDE_DIRECTORY}/aws/pinpoint/model)

if(PLATFORM_WINDOWS AND MSVC)
    file(GLOB NUGET_FILES "nuget/${PROJECT_NAME}.*.autopkg")
    install (FILES ${NUGET_FILES} DESTINATION nuget)
endif()

do_packaging()
